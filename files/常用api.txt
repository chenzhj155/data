// 完整的API注册表
class ComprehensiveApiMonitor {
private:
    std::map<ApiCategory, const char**> apiRegistry;
    
public:
    ComprehensiveApiMonitor() {
        InitializeApiRegistry();
    }
    
    void InitializeApiRegistry() {
        apiRegistry[ApiCategory::SYNCHRONIZATION] = SynchronizationAPIs;
        apiRegistry[ApiCategory::COMPLETION_PORT] = CompletionPortAPIs;
        apiRegistry[ApiCategory::EXCEPTION_HANDLING] = ExceptionAPIs;
        apiRegistry[ApiCategory::SHELL_EXECUTE] = ShellExecuteAPIs;
        apiRegistry[ApiCategory::CRYPTOGRAPHY] = CryptographyAPIs;
        apiRegistry[ApiCategory::RANDOM_NUMBER] = RandomNumberAPIs;
        apiRegistry[ApiCategory::COM_INITIALIZATION] = COMInitializationAPIs;
        apiRegistry[ApiCategory::LATERAL_MOVEMENT] = LateralMovementAPIs;
    }
    
    // 获取所有API用于监控
    std::vector<std::string> GetAllApisForCategory(ApiCategory category) {
        std::vector<std::string> apis;
        auto it = apiRegistry.find(category);
        
        if (it != apiRegistry.end()) {
            const char** apiArray = it->second;
            for (int i = 0; apiArray[i] != nullptr; i++) {
                apis.push_back(std::string(apiArray[i]));
            }
        }
        
        return apis;
    }
    
    // 获取总数统计
    std::map<ApiCategory, size_t> GetApiCounts() {
        std::map<ApiCategory, size_t> counts;
        for (auto& [category, apis] : apiRegistry) {
            size_t count = 0;
            for (int i = 0; apis[i] != nullptr; i++) {
                count++;
            }
            counts[category] = count;
        }
        return counts;
    }
};
